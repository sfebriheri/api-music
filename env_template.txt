# PostgreSQL Database Configuration Template
# Copy this file to .env and update with your database credentials

# Database connection string (preferred method)
DATABASE_URL=postgresql://username:password@localhost:5432/database_name

# Individual database parameters (alternative method)
DB_HOST=localhost
DB_PORT=5432
DB_NAME=your_database_name
DB_USER=your_username
DB_PASSWORD=your_password

# Spotify API Configuration
SPOTIFY_CLIENT_ID=your_client_id_here
SPOTIFY_CLIENT_SECRET=your_client_secret_here

# Music Download Directory (optional)
MUSIC_DOWNLOAD_DIR=~/Music
